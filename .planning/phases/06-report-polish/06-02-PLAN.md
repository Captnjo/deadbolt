1. Create `lib/shared/clipboard_helper.dart`:
   - A top-level utility for clipboard copy with auto-clear:
     ```dart
     import 'dart:async';
     import 'package:flutter/material.dart';
     import 'package:flutter/services.dart';

     /// Global clipboard auto-clear timer. Only one copy at a time.
     Timer? _clipboardClearTimer;

     /// Copy [text] to clipboard, show a SnackBar, and auto-clear after 30 seconds.
     ///
     /// The SnackBar message defaults to 'Copied to clipboard'.
     /// Pass [sensitive] = true to append '(clears in 30s)' to the toast.
     void copyWithAutoClear(
       BuildContext context,
       String text, {
       String message = 'Copied to clipboard',
       bool sensitive = false,
     }) {
       Clipboard.setData(ClipboardData(text: text));
       _clipboardClearTimer?.cancel();
       _clipboardClearTimer = Timer(const Duration(seconds: 30), () {
         Clipboard.setData(const ClipboardData(text: ''));
       });
       final displayMessage = sensitive ? '$message (clears in 30s)' : message;
       ScaffoldMessenger.of(context).showSnackBar(
         SnackBar(
           content: Text(displayMessage),
           duration: const Duration(seconds: 2),
         ),
       );
     }

     /// Cancel any pending clipboard clear timer. Call from dispose() if needed.
     void cancelClipboardClear() {
       _clipboardClearTimer?.cancel();
       _clipboardClearTimer = null;
     }
     ```
